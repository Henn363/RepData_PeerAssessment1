data <- mutate(data, DOW = wday(data$date, label = TRUE))

Avg_wk_final <- melt(Avg_wk, id.vars = c("Factor", "interval"))
Avg_wk_final <- dcast(Avg_wk_final, Factor + interval ~ variable, fun = mean)
